#include <iostream>
using namespace std;
//overload = menimpa(contoh disini kita menggunakan macam-macam luas persegi)
int luas_persegi(int p, int l){//fungsi dasar
    int luas = p * l;
    return luas;
}
int luas_persegi(int p){//fungsi overload
    int luas = p * p;
    return luas;
}
double luas_persegi(double p, double l){//fungsi overload
    double luas = p * l;
    return luas;
}
int main()
{
    cout<<"luas persegi: "<<luas_persegi(3,2)<<endl;
    cout<<"luas persegi: "<<luas_persegi(3)<<endl;
    cout<<"luas persegi: "<<luas_persegi(2.5,4.3)<<endl;
    return 0;
}

#include <iostream>
using namespace std;

void fungsi (int &);
void kuadrat (int &);

int main()
{
    int a = 5 ;
    cout<<"adress a: "<<&a<<endl;
    cout<<"nilai a: "<<a<<endl<<endl;
    fungsi(a);
    kuadrat(a);
    return 0;
}

    cout<<"adress b: "<<&b<<endl;

}

void kuadrat(int &nilai){
    nilai= nilai * nilai;
    cout<<"hasil kuadrat: "<<nilai<<endl;
}

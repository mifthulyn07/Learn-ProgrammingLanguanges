#include <iostream>
using namespace std;

union data{
    int a;
    char b[5];
};
/*jika struct 1 data punya banyak komponen,
maka union 1 data bisa banyak bentuk*/
int main()
{
    data otong;
    otong.a = 12345;
    cout<<"data a: "<<otong.a<<endl;
    cout<<"data b: "<<otong.b<<endl<<endl;
    otong.b[0] = 'a';
    otong.b[1] = 'b';
    otong.b[2] = 'c';
    otong.b[3] = 'd';

    cout<<"data a: "<<otong.a<<endl;
    cout<<"data b: "<<otong.b<<endl;

    return 0;
}

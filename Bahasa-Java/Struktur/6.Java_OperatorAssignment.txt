package miftahul_ulyana_hutabarat;
public class Miftahul_Ulyana_Hutabarat {
    public static void main(String[] args) {
      //assignment
      
      //assignment penjumlahan
      int a = 10;
      a += 5; //a = a + 5
      System.out.println("nilai penjumlahan adalah: " + a);
      
      //assignment pengurangan
      int b = 10;
      b -= 5; //b = b - 5
      System.out.println("nilai pengurangan adalah: " + b);
      
      //assignment perkalian
      int c = 10;
      c *= 5; //c = c * 5
      System.out.println("nilai perkalian adalah: " + c);
      
      //assignment pembagian
      int d = 10;
      d /= 5; //d = d / 5
      System.out.println("nilai pembagian adalah: " + d);
      
      //assignment modulus
      int e = 11;
      e %= 5; 
      System.out.println("nilai modulus adalah: " + e);
    }
}

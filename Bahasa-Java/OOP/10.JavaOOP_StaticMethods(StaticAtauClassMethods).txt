package Mifthulyn07;
import java.util.ArrayList;

class Player{
    private String name;
    static int numberOfPlayer;
    private static ArrayList<String> nameList = new ArrayList<String>();
    
    
    Player(String name){
        this.name = name;
        Player.numberOfPlayer++;
        Player.nameList.add(this.name);//isi semua nama dari player 
    }
    
    void show(){
        System.out.println("player name = " + this.name);
    }
    
    //static method
    static void shownumberOfPlayer(){
        System.out.println("number of player = " + Player.numberOfPlayer);
    }
    
    static ArrayList<String> getnames(){
        return Player.nameList;
    }
}

public class MiftahulUlyanaHutabarat {
    public static void main(String[] args) {
       Player player1 = new Player("Nurli");
       Player player2 = new Player("Putri");
       Player player3 = new Player("Miftah");
       Player player4 = new Player("lala");
       
       Player.shownumberOfPlayer();
       
       System.out.println("names = " + Player.getnames());//alternatif1
//       System.out.println("names = " + player1.getnames());//alternatif2(tidak recomended)
    }
}

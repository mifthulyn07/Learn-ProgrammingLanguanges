#include <iostream>
using namespace std;

void printArray(int *ptrArray, int baris, int kolom){//menentukan baris = 2, dan kolom = 3
    int index=0;
    for(int i=0; i<kolom; i++){
        cout<<" [ ";
        for(int j=0; j<baris; j++){
            cout<<*(ptrArray + index)<<" ";
            index++;
        }
        cout<<"]"<<endl;
    }
}

int main()
{
    int nilai[2][3]={1,2,3,4,5,6};//array multidimensi
    printArray(*nilai, 2, 3);
    return 0;
}

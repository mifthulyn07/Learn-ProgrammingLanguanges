#include <iostream>
using namespace std;
int main(){
    int k[1000], i, n, jlh=0;
    cout<<"Banyaknya Jumlah Elemen Array= ";cin>>n;
    for(i=1; i<=n; i++){
        cout<<"Elemen Array ["<<i<<"] =";cin>>k[i];
        jlh=jlh+k[i];
    }
    cout<<"jumlah dari elemen array"<<endl;
    for(i=1; i<=n; i++){
        cout<<k[i];
        if(i<n){
            cout<<"+";
    }
    }
    cout<<endl;
    cout<<"jumlah: "<<jlh;
    return 0;
}

#include<iostream>
using namespace std;
int main()
{
    int nilai[100],bintang;
    cout<<"program menampilkan grafik nilai"<<endl<<endl;
    for(int i=0; i<=10; i++){
        cout<<"jumlah mahasiswa dengan nilai: ";
        if(i==0){
            cout<<"0-9  : ";
        }
        else if(i==10){
            cout<<"100  : ";
        }
        else {
            cout<<i*10<<"-"<<i*10+9<<": ";
        }
        cin>>nilai[i];
    }
    cout<<endl;
    cout<<"grafik nilai"<<endl<<endl;
    for(int i=0; i<=10; i++){
        if(i==0){
            cout<<"0-9  : ";
        }
        else if(i==10){
            cout<<"100  : ";
        }
        else {
            cout<<i*10<<"-"<<i*10+9<<": ";
        }
        for(bintang=1; bintang<=nilai[i]; bintang++){
            cout<<"*";
        }
        cout<<endl;
    }

    return 0;
}

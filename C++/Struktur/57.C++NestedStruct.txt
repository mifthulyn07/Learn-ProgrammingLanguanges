#include <iostream>
#include <string>
using namespace std;

struct aktor{
    string nama;
    int tahun_kelahiran;
};

struct film{
    string genre;
    int tahun;
    aktor pemeran1;//nesting struct(memasukkan struct aktor ke dalam struct film)
    aktor pemeran2;
};

int main()
{
    aktor pemain1, pemain2;
    film cintabertasbih;

    pemain1.nama = "miftahul";
    pemain1.tahun_kelahiran = 2001;

    pemain2.nama = "ulyana";
    pemain2.tahun_kelahiran = 2001;

    cintabertasbih.genre = "romance";
    cintabertasbih.tahun = 2009;
    cintabertasbih.pemeran1 = pemain1;
    cintabertasbih.pemeran2 = pemain2;

    cout<<"ketika cinta bertasbih"<<endl;
    cout<<"genre: "<<cintabertasbih.genre<<endl;
    cout<<"tahun: "<<cintabertasbih.tahun<<endl;
    cout<<"pemeran 1: "<<cintabertasbih.pemeran1.nama<<endl;
    cout<<"pemeran 2: "<<cintabertasbih.pemeran2.nama<<endl;
    return 0;
}

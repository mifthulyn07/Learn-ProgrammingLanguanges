#include <iostream>
using namespace std;
int fibonacci(int n);
int main()
{
    int n;
    cout<<"masukkan banyaknya fibonacci: ";cin>>n;
    cout<<"hasil dari fionacci: "<<fibonacci(n)<<endl;
    return 0;
}
int fibonacci(int n){
    if((n==0) || (n==1)){
        return n;
    }
    else{
        return fibonacci(n-1) + fibonacci(n-2);
    }
}
